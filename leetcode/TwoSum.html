(0001) -> { Two Sum }

Description

Given an array of integers, return indices of the two numbers such that they add up to a specific target.

You may assume that each input would have exactly one solution, and you may not use the same element twice.

Example:

Given nums = [2, 7, 11, 15], target = 9,

Because nums[0] + nums[1] = 2 + 7 = 9, return [0, 1].

Rust Solution


# #![allow(unused_variables)]
#fn main() {
use std::convert::TryFrom;
use std::{collections::HashMap, vec};

pub fn two_sum(nums: Vec<i32>, target: i32) -> Vec<i32> {
    let mut element_index_cache: HashMap<i32, i32> = HashMap::new();

    for (i, e) in nums.into_iter().enumerate() {
        let key = target - e;
        match element_index_cache.get(&key) {
            Some(v) => {
                return vec![*v, i as i32];
            }
            None => {
                element_index_cache.insert(e, i as i32);
            }
        }
    }

    panic!("impossible")
}

pub fn two_sum_brute_force(nums: Vec<i32>, target: i32) -> Vec<i32> {
    for (i, e1) in nums.iter().enumerate() {
        for (j, e2) in nums.iter().enumerate() {
            if i != j && *e1 + *e2 == target {
                return vec![i as i32, j as i32];
            }
        }
    }

    panic!("impossible")
}


pub fn two_sum_rec(nums: Vec<i32>, target: i32) -> Vec<i32> {
    fn func<'a>(
        nums: &Vec<i32>,
        target: i32,
        element_index_cache: &'a mut HashMap<i32, i32>,
        len: usize,
    ) -> (&'a HashMap<i32, i32>, Option<Vec<i32>>) {
        match len {
            0 => (element_index_cache, None),
            _ => {
                let curr_index = len - 1;
                let curr_index_i32 = i32::try_from(curr_index).unwrap();
                let curr_value = nums[curr_index];
                let another = target - curr_value;

                match element_index_cache.get(&another) {
                    Some(v) => (element_index_cache, Some(vec![curr_index_i32, *v])),
                    None => {
                        element_index_cache.insert(curr_value, curr_index_i32);
                        func(nums, target, element_index_cache, len - 1)
                    }
                }
            }
        }
    }

    match func(&nums, target, &mut HashMap::new(), nums.len()) {
        (_, Some(rst)) => rst,
        _ => panic!("impossible"),
    }
}

package kleetcode.k0001
/*
在想这道题之前，先做一道更简单的。
Given an array of integers, return indice of the number equal to a specfic target.

Example:

Given nums = [2, 7, 11, 15], target = 7
Because nums[1] = 7 == target, return 1

为了验证条件是否满足，程序需要访问每一个长度为 1 的单元的信息来判断。
程序每次的视域只能为常数个单元，不妨设置常数为 1
访问 长度 为 N 的 Array 需要视域 为 1 的 程序访问 N 次
每一次访问可以 通过验证 num == T 来获得唯一的确认。
可以推断 遍历一遍 num 即可获得所需结果。
时间复杂度为O(N)

现在的验证方式发生了改变。验证本身依然需要不可避免的时间复杂度 O(N)
需要验证 num + ? == T 来获得唯一的确认
也即 ？== T - num, 其中 ? 在 nums 范围内。
问题退化为我们在上面解决过的问题。
最佳时间复杂度为 O(N)

如果，使用 O(N) 的时间 构建缓存服务
O(N)[build cache] + O(N) (filter) * O(1) (predicate) 
|> O(N)

但是每次执行函数，我们总是需要预先使用 O(N) 的时间 O(N) 的空间 来构建
<curr value，curr index> 的 cache

如果我们可以在执行 filter 的同时构建缓存的话，就可以减少平均情况下的时间和空间消耗
当前 索引 每次将 array 分成 两部分。
我们最多在执行 filter 的时候 将当前索引之前的所有元素建立缓存。
可以容易的想到 只要在 最后一个元素之前构建完之前所有元素的索引即可找到最后一个是否符合条件
cache = {2, 7, 5} | verify 13
如果 13 在之前的缓存中找不到 满足条件的，
可以安全的缩小当前 array 为 [2, 7, 5]

*/
fun twoSum(nums: IntArray, target: Int): IntArray {
    val map = hashMapOf<Int, Int>()

    for (i in 0 until nums.size) {

        if (map.containsKey(target - nums[i])) {
            val tmp = map[target - nums[i]]!!.toInt()
            return intArrayOf(tmp, i)
        }

        map[nums[i]] = i
    }
    return intArrayOf()
}

// try.kotlinlang.org
fun main() {
    println(twoSum(intArrayOf(2, 7, 15, 13), 9).joinToString { e -> e.toString() })
}

package main

import (
	"fmt"

/*TwoSum defination
 */
func TwoSum(nums []int, target int) []int {
	elementIndexCache := map[int]int{}
	
	for i, e := range nums {
		key := target - e
		
		if v, ok := elementIndexCache[key]; ok {
			return []int{v, i}
		}

		elementIndexCache[e] = i
	}

	panic("impossible")
}

func main() {
	fmt.Printf("%v\n", TwoSum([]int{2, 7, 11, 15}, 9))
}

import unittest
from typing import List
from collections import deque


def CanFinish(coursesCount: int, preRequisites: List[List[int]]) -> bool:
    adjOut = [set() for _ in range(coursesCount)]
    for [a, b] in preRequisites:
        adjOut[b].add(a)

    adjIn = [set() for _ in range(coursesCount)]
    for [a, b] in preRequisites:
        adjIn[a].add(b)

    S = deque()

    for i in range(coursesCount):
        if len(adjIn[i]) == 0:
            S.append(i)

    while len(S) != 0:
        n = S.pop()
        for m in adjOut[n]:
            adjIn[m].remove(n)
            if len(adjIn[m]) == 0:
                S.append(m)

    edgesCount = 0

    for e in adjIn:
        edgesCount += len(e)

    return edgesCount == 0


if __name__ == "__main__":
    print(CanFinish(0, [[]]))


class TestCanFinish(unittest.TestCase):
    def test(self):
        self.assertTrue(CanFinish(2, [[1, 0]]))
        self.assertTrue(CanFinish(4, [[1, 0], [2, 0], [3, 1], [3, 2]]))
        self.assertFalse(CanFinish(2, [[1, 0], [0, 1]]))

export const WordBreak = (str: string, dict: Set<string>): boolean => {
  const dp: Array<boolean> = Array(str.length + 1).fill(false);
  dp[0] = true;

  for (let i = 1; i < str.length + 1; i++) {
    for (let j = 0; j < i; j++) {
      const substr = str.substring(j, i);

      if (dp[j] && dict.has(substr)) {
        dp[i] = true;
        break;
      }
    }
  }
  return dp[str.length];
};

const WordBreak2: (str: string, dict: Set<string>) => boolean = F.Memoize(
  (str: string, dict: Set<string>): boolean => {

    for (let i = 0; i < str.length; i++) {
      const x = str.substring(0, i);
      const xs = str.substring(i, str.length);
      if (WordBreak2(x, dict) && WordBreak2(xs, dict)) {
        return true;
      }
    }
    return false;
  },
  (args) => (JSON.stringify([args[0], ...args[1]])),
